\subsection*{19. Fragen zum Kompetenzerwerb}

\tikzmark{f4a} % Auskommentieren, falls keine Trennlinie

\headerfive{6.7cm}{\small trifft voll zu}{\small trifft überwiegend zu}{\small trifft weniger zu}{\small trifft gar nicht zu}{keine Angabe} \\

\vspace{-.5cm}

\begin{enumerate}

\textbf{Mit Fachwissen umgehen} \\ \\
\itemsfive{ke_1}{6cm}{\item Das vermittelte und erworbene Fachwissen ist als Grundlagenwissen relevant für das Lehramt an Grundschulen.}{1}{2}{3}{4}{-1} \\
\itemsfive{ke_2}{6cm}{\item Das Fachwissen stellt eine Unterstützung bei der Unterrichtsvorbereitung im Fach Naturwissenschaften in der Grundschule dar.}{1}{2}{3}{4}{-1} \\
\itemsfive{ke_3}{6cm}{\item Das Fachwissen entsprechend der Rahmenlehrpläne der SEK-I gibt Sicherheit für den Unterricht in der Grundschule.}{1}{2}{3}{4}{-1} \\
\vspace{.1cm}
\textbf{Erkenntnisgewinn} \\ \\
\itemsfive{ke_4}{6cm}{\item Es werden verschiedene naturwissenschaftliche Untersuchungsmethoden wie das Beobachten, Vergleichen, Ordnen und Experimentieren durchgeführt.}{1}{2}{3}{4}{-1} \\
\itemsfive{ke_5}{6cm}{\item Es werden Denk- und Argumentationsweisen wie das Bilden von Hypothesen und das Herstellen von Zusammenhängen (beispielsweise Wenn-dann-Beziehungen, Proportionalitäten) vermittelt.}{1}{2}{3}{4}{-1} \\
\itemsfive{ke_6}{6cm}{\item Es wird mit Geräten und Chemikalien sachgerecht umgegangen.}{1}{2}{3}{4}{-1} \\
\itemsfive{ke_7}{6cm}{\item Es werden Elemente der Mathematik beim Umgang mit Größen, dem Erfassen von Messwerten und dem Anwenden einfacher mathematischer Verfahren berücksichtigt.}{1}{2}{3}{4}{-1} \\
\itemsfive{ke_8}{6cm}{\item Die Arbeit mit Modellen wird durchgeführt.
}{1}{2}{3}{4}{-1} \\
\vspace{.05cm}
\textbf{Kommunikation} \\ \\
\itemsfive{ke_9}{6cm}{\item Zur Beschreibung von Phänomenen aus Natur und Technik wird auf die Nutzung der Fachsprache Wert gelegt.}{1}{2}{3}{4}{-1} \\
\itemsfive{ke_10}{6cm}{\item Es werden Sachtexte, grafische Darstellungen, Modelle und andere Medien in die Veranstaltungen einbezogen.}{1}{2}{3}{4}{-1} \\
\textbf{Bewertung} \\ \\
\itemsfive{ke_11}{6cm}{\item Fachinformationen werden in den Veranstaltungen bewertet und Erklärungen nachvollzogen und reflektiert.}{1}{2}{3}{4}{-1}
